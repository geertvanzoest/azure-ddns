```bash
#!/bin/bash
# Test helper voor ns4j bats tests
# Source ns4j functies zonder main() uit te voeren (BASH_SOURCE guard)

# Pad naar het ns4j script (relatief vanuit test/ directory)
NS4J_SCRIPT="${BATS_TEST_DIRNAME}/../ns4j"

# Source het script -- BASH_SOURCE guard voorkomt main() executie
load_ns4j() {
    # Zet minimale env vars zodat source niet faalt op set -u
    export AZURE_TENANT_ID="${AZURE_TENANT_ID:-test-tenant}"
    export AZURE_CLIENT_ID="${AZURE_CLIENT_ID:-test-client}"
    export AZURE_CLIENT_SECRET="${AZURE_CLIENT_SECRET:-test-secret}"
    export AZURE_SUBSCRIPTION_ID="${AZURE_SUBSCRIPTION_ID:-test-sub}"
    export AZURE_RESOURCE_GROUP="${AZURE_RESOURCE_GROUP:-test-rg}"
    export DNS_ZONE_NAME="${DNS_ZONE_NAME:-test.example.com}"
    export DNS_RECORD_NAME="${DNS_RECORD_NAME:-home}"

    source "$NS4J_SCRIPT"
}

# Curl mock: vervang curl met een functie die gecontroleerde output geeft
# Gebruik: setup_curl_mock "response_body" "http_code"
setup_curl_mock() {
    local body="$1"
    local http_code="${2:-200}"

    # Exporteer mock functie die curl vervangt
    eval "curl() { echo '${body}'; echo '${http_code}'; }"
    export -f curl
}

# Curl mock voor meerdere calls (fallback testing)
# Gebruik: setup_curl_sequence "body1|code1" "body2|code2"
setup_curl_sequence() {
    local -a responses=("$@")
    local call_count_file
    call_count_file=$(mktemp)
    echo "0" > "$call_count_file"

    # Bouw de mock functie dynamisch
    local mock_body="local idx; idx=\$(cat ${call_count_file}); "
    local i=0
    for resp in "${responses[@]}"; do
        local body="${resp%%|*}"
        local code="${resp##*|}"
        mock_body+="if [[ \$idx -eq $i ]]; then echo '$body'; echo '$code'; echo \$((\$idx + 1)) > ${call_count_file}; return 0; fi; "
        i=$((i + 1))
    done
    # Als alle responses op zijn, faal
    mock_body+="return 1; "

    eval "curl() { $mock_body }"
    export -f curl
    echo "$call_count_file"
}

# Cleanup
teardown_curl_mock() {
    unset -f curl 2>/dev/null || true
}
```

De helper biedt drie kernfuncties:
- `load_ns4j`: source het script met dummy env vars (set -u safe)
- `setup_curl_mock`: simpele mock voor single-response tests
- `setup_curl_sequence`: mock voor fallback-cascade tests (meerdere calls)

**test/test_ip_fallback.bats (IP-04):**
```bats
#!/usr/bin/env bats

setup() {
    load test_helper
    load_ns4j
}

teardown() {
    teardown_curl_mock
}

@test "IP-04: eerste service succesvol, tweede niet nodig" {
    setup_curl_mock "93.184.216.34" "200"
    run get_public_ip
    [ "$status" -eq 0 ]
    [ "$output" = "93.184.216.34" ]
}

@test "IP-04: eerste service faalt (HTTP 500), fallback naar tweede" {
    local tmpfile
    tmpfile=$(setup_curl_sequence "error|500" "93.184.216.34|200")
    run get_public_ip
    [ "$status" -eq 0 ]
    [[ "$output" =~ "93.184.216.34" ]]
    rm -f "$tmpfile"
}

@test "IP-04: eerste service geeft ongeldig IP, fallback naar tweede" {
    local tmpfile
    tmpfile=$(setup_curl_sequence "not-an-ip|200" "10.0.0.1|200")
    run get_public_ip
    [ "$status" -eq 0 ]
    [[ "$output" =~ "10.0.0.1" ]]
    rm -f "$tmpfile"
}

@test "IP-04: alle services falen, exit code 2 (EXIT_IP)" {
    local tmpfile
    tmpfile=$(setup_curl_sequence "error|500" "error|503")
    run get_public_ip
    [ "$status" -eq 2 ]
    [[ "$output" =~ "Geen geldige IP-service beschikbaar" ]]
    rm -f "$tmpfile"
}
```

**test/test_locking.bats (OPS-03):**
```bats
#!/usr/bin/env bats

# flock is alleen beschikbaar op Linux (util-linux)
# Deze tests worden overgeslagen op macOS

setup() {
    if [[ "$(uname)" != "Linux" ]]; then
        skip "flock niet beschikbaar op $(uname) -- alleen Linux"
    fi
    load test_helper
}

@test "OPS-03: LOCK_FILE is /tmp/ns4j.lock" {
    load_ns4j
    [ "$LOCK_FILE" = "/tmp/ns4j.lock" ]
}

@test "OPS-03: script bevat flock --nonblock" {
    grep -q "flock --nonblock" "$NS4J_SCRIPT"
}

@test "OPS-03: geblokkeerde lock geeft WARN en exit 0" {
    # Simuleer een geblokkeerde lock door het script tweemaal te starten
    # Eerste instantie houdt lock vast, tweede moet WARN geven
    exec 200>/tmp/ns4j.lock
    flock --nonblock 200

    # Start tweede instantie (moet falen op flock)
    # Zet dummy env vars
    export AZURE_TENANT_ID="test" AZURE_CLIENT_ID="test" AZURE_CLIENT_SECRET="test"
    export AZURE_SUBSCRIPTION_ID="test" AZURE_RESOURCE_GROUP="test"
    export DNS_ZONE_NAME="test.com" DNS_RECORD_NAME="test"

    run bash "$NS4J_SCRIPT"
    [ "$status" -eq 0 ]
    [[ "$output" =~ "Andere instantie draait" ]]

    # Release lock
    flock --unlock 200
    exec 200>&-
}
```
