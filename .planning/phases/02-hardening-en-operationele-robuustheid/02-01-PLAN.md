**1. Vervang de complete get_public_ip() functie door de fallback-cascade versie:**
```bash
get_public_ip() {
    local ip http_code response body
    for service in "${IP_SERVICES[@]}"; do
        response=$(curl --silent --show-error --max-time 10 \
            --write-out "\n%{http_code}" "$service" 2>/dev/null) || {
            debug "Service ${service} niet bereikbaar"
            continue
        }
        http_code=$(echo "$response" | tail -n1)
        body=$(echo "$response" | sed '$d')
        ip=$(echo "$body" | tr -d '[:space:]')

        if [[ "$http_code" != "200" ]]; then
            debug "Service ${service}: HTTP ${http_code}"
            continue
        fi
        if [[ ! "$ip" =~ $IP_REGEX ]]; then
            debug "Service ${service}: ongeldig IP '${ip}'"
            continue
        fi
        debug "Service ${service}: IP=${ip}"
        echo "$ip"
        return 0
    done
    log "ERROR" "Geen geldige IP-service beschikbaar (geprobeerd: ${IP_SERVICES[*]})"
    exit "${EXIT_IP}"
}
```
Per D-01: cascade icanhazip -> checkip.amazonaws, eerste geldige wint. Per D-02: inline IPv4-regex validatie na elke response. Per D-03: log alle geprobeerde URLs en exit 2 bij uitputting. BELANGRIJK: `$IP_REGEX` is UNQUOTED in de `=~` vergelijking (anders matcht het als literal string). De `|| { continue; }` na curl voorkomt dat `set -e` het script stopt bij een onbereikbare service.

```bash
main() {
    parse_args "$@"
    validate_config

    local access_token
    access_token=$(get_access_token)

    local public_ip
    public_ip=$(get_public_ip)

    local current_ip
    current_ip=$(get_dns_record "$access_token")

    if [[ "$current_ip" == "$public_ip" ]] && [[ "$FORCE" -eq 0 ]]; then
        log "INFO" "IP ongewijzigd (${public_ip})"
        exit "${EXIT_OK}"
    fi

    if [[ "$FORCE" -eq 1 ]]; then
        debug "Force modus: update ongeacht IP-vergelijking"
    fi

    if [[ -z "$current_ip" ]]; then
        log "INFO" "Geen bestaand A-record gevonden, wordt aangemaakt"
    else
        log "INFO" "IP gewijzigd: ${current_ip} -> ${public_ip}"
    fi

    update_dns "$access_token" "$public_ip"

    exit "${EXIT_OK}"
}
```
